#include <stdio.h>

int main() {
    int minutes, hours, remMinutes;

    printf("Enter total minutes: ");
    scanf("%d", &minutes);

    hours = minutes / 60;
    remMinutes = minutes % 60;

    printf("Hours = %d\n", hours);
    printf("Minutes = %d\n", remMinutes);

    return 0;
}

output:

  Enter total minutes: 130
  Hours = 2
  Minutes = 10
